\tikzset{
  greybox/.style={draw=black!60, fill=gray!20, thin,
             minimum height=0.6cm, inner sep=2pt,
             font=\small, align=center},
  yellowbox/.style={draw=black!60, fill=yellow!60, thin,
             minimum height=0.6cm, inner sep=2pt,
             font=\small, align=center},
  greenbox/.style={draw=black!60, fill=green!25, thin,
             minimum height=0.6cm, inner sep=2pt,
             font=\small, align=center},
  corelabel/.style={font=\small\bfseries, anchor=east},
  timearrow/.style={-{Stealth[length=6pt]}, thick},
  waitarrow/.style={-{Stealth[length=6pt]}, dashed, gray, thick},
  prockarrow/.style={-{Stealth[length=6pt]}, dotted, black!70, thick},
  dashedline/.style={dashed, gray},
  nowait/.style={circle, draw=red!70!black, fill=red!15,

}

% Usage: \Block{xleft}{ycenter}{width}{style}{text}
\newcommand{\Block}[5]{%
  \node[#4, minimum width=#3 cm] at ({#1+#3/2},{#2}) {#5};}

\newcommand{\WaitArrow}[4]{%
  \draw[waitarrow] (#1,#2) --
    node[above, font=\scriptsize, gray, midway]{#4} (#3,#2);}

\newcommand{\Async}[3]{
 \draw[dashedline] (#1,#2) --(#3,#2);}

\newcommand{\ProcArrow}[4]{%
  \draw[prockarrow] (#1,#2) --
    node[above, font=\scriptsize, black!70, midway]{#4} (#3,#2);}

\newcommand{\ZFlushAnnot}[4]{%
  \def\ztop{#3+1}%
  \draw[gray]
    ({#1},{\ztop}) -- ({#2},{\ztop})
    -- ({#2},{#3+0.75});
    \draw[gray, -{Stealth[length=4pt]}]
    ({#1},{\ztop}) -- ({#1},{#3+0.4});
  \node[font=\scriptsize, align=center]
    at ({(#1+#2)/2},{#3+1})
    {Flush for $z$\\iterations};
    \node[font=\scriptsize, align=center]
    at ({#2},{#3+0.6})
    {comp+=#4};}


\begin{document}

\def\we{1.20}   % width enQ
\def\wc{1.20}   % width copy from queue
\def\wi{3.2}    % width flush all
\def\wf{1.20}   % width free iova
\def\wcb{1.20}  % width execute callback
\def\wp{1.20}   % width free pfn
\def\gap{0.1}
\def\bgap{0.40}
\def\asyncgap{0.9}
\def\proclen{3.0}

\def\rA{5.5}
\def\rB{4}
\def\rC{2}
\def\rD{0.0}

% ── C0 leader round 1 positions ──
\pgfmathsetmacro\AeX{0.40}
\pgfmathsetmacro\AcX{\AeX+\we+\gap}
\pgfmathsetmacro\AiX{\AcX+\wc+\gap}
\pgfmathsetmacro\AfX{\AiX+\wi+\gap}
\pgfmathsetmacro\AcbX{\AfX+\wf+\gap}
\pgfmathsetmacro\AZend{\AcbX+\wcb+\bgap}

% ── C1 follower enqueue ──
\pgfmathsetmacro\BeX{\AiX+\bgap}
\pgfmathsetmacro\Bnw{\BeX+\we+1.6}

% ── C2 follower enqueue ──
\pgfmathsetmacro\CeX{\BeX+\bgap*3}
\pgfmathsetmacro\Cnw{\CeX+\we+1.6}

% ── C3 enqueue + becomes leader round 2 ──
\pgfmathsetmacro\DeX{\CeX+\bgap*6}
\pgfmathsetmacro\DcX{\AZend+\gap}
\pgfmathsetmacro\DiX{\DcX+\wc+\gap}
\pgfmathsetmacro\DfX{\DiX+\wi+\gap}
\pgfmathsetmacro\DcbX{\DfX+\wf+\gap}
\pgfmathsetmacro\DZend{\DcbX+\wcb+\bgap}
\pgfmathsetmacro\DpX{\DcbX+\wcb+\asyncgap}

% ── Round 2 enqueues ──
\pgfmathsetmacro\AeSecX{\AcbX+\wcb+\asyncgap+\bgap}
\pgfmathsetmacro\BeSecX{\AeSecX+\bgap*3}
\pgfmathsetmacro\CeSecX{\BeSecX+\bgap*3}

\pgfmathsetmacro\Xtotal{\DpX+\wp+0.60}

\begin{tikzpicture}

  % ── Core labels ──
  \node[corelabel] at (0,\rA) {C0};
  \node[corelabel] at (0,\rB) {C1};
  \node[corelabel] at (0,\rC) {C2};
  \node[corelabel] at (0,\rD) {C3};

  % ══════════════════════════════════════
  % C0: leader round 1
  % ══════════════════════════════════════
  \Block{\AeX}{\rA}{\we}{yellowbox}{enQ}
  \Block{\AcX}{\rA}{\wc}{yellowbox}{copy\\from Q}
  \Block{\AiX}{\rA}{\wi}{greybox}{IOTLB flush\\all}
  \Block{\AfX}{\rA}{\wf}{greybox}{Free\\iovas}
  \Block{\AcbX}{\rA}{\wcb}{greenbox}{exe\\fp cb}

  % annotations for C0
  \node[font=\scriptsize, gray, align=center]
    at ({\AeX+\we/2},{\rA+1.1}) {comp$=0$};
  \node[font=\scriptsize, gray, align=center]
    at ({\AeX+\we/2},{\rA+0.8}) {set BUSY};
  \node[font=\scriptsize, gray, align=center]
    at ({\AeX+\we/2},{\rA+0.5}) {seq$=1$};
  \node[font=\scriptsize, gray, align=center]
    at ({\AcbX+\wcb+\asyncgap/2+0.1},{\rA+0.5}) {unset\\BUSY};
  \Async{\AcbX+\wcb}{\rA}{\AcbX+\wcb+\asyncgap}

  \ZFlushAnnot{\AcX+\wc/2}{\AfX+\wf/2}{\rA}{1}

  % ══════════════════════════════════════
  % C1: follower — enqueue, NO WAIT, process packets
  % ══════════════════════════════════════
  \Block{\BeX}{\rB}{\we}{yellowbox}{enQ}
  \node[font=\scriptsize, gray]
    at ({\BeX+\we/2},{\rB+0.55}) {seq$=2$};

  % short dashed to no-wait symbol
  \draw[waitarrow] ({\BeX+\we+\gap},\rB) -- ({\Bnw-0.22},\rB);
  \node[font=\scriptsize, gray, align=center]
    at ({(\BeX+\we+\gap+\Bnw-0.22)/2},{\rB+0.4}) {wait if comp $<$ seq$=2$};

  \draw[red!70!black, line width=1.2pt]
    ([shift={(-0.12,0.12)}]B1nw.center) -- ([shift={(0.12,-0.12)}]B1nw.center);

  % process other packets
  \ProcArrow{\Bnw+0.25}{\rB}{\Bnw+\proclen}{Process other packets}

  % second round enqueue
  \Block{\BeSecX}{\rB}{\we}{yellowbox}{enQ}
  \node[font=\scriptsize, gray]
    at ({\BeSecX+\we/2},{\rB+0.55}) {seq$=6$};

  % ══════════════════════════════════════
  % C2: follower — enqueue, NO WAIT, process packets
  % ══════════════════════════════════════
  \Block{\CeX}{\rC}{\we}{yellowbox}{enQ}
  \node[font=\scriptsize, gray]
    at ({\CeX+\we/2},{\rC+0.55}) {seq$=3$};

  % short dashed to no-wait symbol
  \draw[waitarrow] ({\CeX+\we+\gap},\rC) -- ({\Cnw-0.22},\rC);
  \node[font=\scriptsize, gray, align=center]
    at ({(\CeX+\we+\gap+\Cnw-0.22)/2},{\rC+0.4}) {wait if comp $<$ seq$=3$};

  % no-wait symbol
  \node[nowait] (C1nw) at (\Cnw,\rC) {};
  \draw[red!70!black, line width=1.2pt]
    ([shift={(-0.12,0.12)}]C1nw.center) -- ([shift={(0.12,-0.12)}]C1nw.center);

  % process other packets
  \ProcArrow{\Cnw+0.25}{\rC}{\Cnw+\proclen}{Process other packets}

  % second round enqueue
  \Block{\CeSecX}{\rC}{\we}{yellowbox}{enQ}
  \node[font=\scriptsize, gray]
    at ({\CeSecX+\we/2},{\rC+0.55}) {seq$=7$};

  % ══════════════════════════════════════
  % C3: enqueue, contend, become leader round 2
  % ══════════════════════════════════════
  \Block{\DeX}{\rD}{\we}{yellowbox}{enQ}
  \node[font=\scriptsize, gray]
    at ({\DeX+\we/2},{\rD+0.55}) {seq$=4$};

  \WaitArrow{\DeX+\we+\gap}{\rD}{\DcX}{Contend to set BUSY}

  \Block{\DcX}{\rD}{\wc}{yellowbox}{copy\\from Q}
  \Block{\DiX}{\rD}{\wi}{greybox}{IOTLB flush\\all}
  \Block{\DfX}{\rD}{\wf}{greybox}{Free\\iovas}

    at ({\DcbX+\wcb+\asyncgap/2+0.1},{\rD+0.5}) {unset\\BUSY};

  % ══════════════════════════════════════
  % C0 round 2: enqueue
  % ══════════════════════════════════════
  \Block{\AeSecX}{\rA}{\we}{yellowbox}{enQ}
